Create Kleur Object

imageObj.onload = function(e) {
    // Create Object of Kleur
    Kleur = new Kleur();
              
    // Pass the image to setup the swatches
    let colorArr = Kleur.setup(imageObj);
              
    // get array of colors and their values
    let array_of_colors = Kleur.generateColorArray(colorArr);
    console.log(array_of_colors);

    // generates random swatch everytime its called
    const vib = Kleur.getSwatch(array_of_colors);
              
    // get light and dark color array using the generated swatch
    let light = Kleur.getLightColors(vib);
    let dark = Kleur.getDarkColors(vib);
};
